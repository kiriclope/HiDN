class CenterScaler:
    """Center (and optionally scale) along trial axis."""
    def __init__(self, axis=0, scale=False):
        self.axis = axis
        self.scale = scale
        self.center_ = None
        self.scale_ = None

    def fit(self, X):
        self.center_ = np.nanmean(X, axis=self.axis, keepdims=True)
        if self.scale:
            self.scale_ = np.nanstd(X, axis=(self.axis, -1), keepdims=True)
            lo, hi = np.nanpercentile(self.scale_.ravel(), [5, 95])
            self.scale_ = np.clip(self.scale_, lo, hi)
        return self

    def transform(self, X):
        out = X - self.center_
        if self.scale and self.scale_ is not None:
            out = out / self.scale_
        return out

    def fit_transform(self, X):
        return self.fit(X).transform(X)

def pad_to_common_neurons(arrays, n_neurons, neuron_offsets):
    """Embed each mouse's array into a common neuron space with NaN padding."""
    padded = []
    for X, offset in zip(arrays, neuron_offsets):
        P = np.full((X.shape[0], n_neurons, X.shape[-1]), np.nan)
        P[:, offset:offset + X.shape[1], :] = X
        padded.append(P)
    return padded

def get_levels(y, factors):
    if isinstance(factors, str):
        factors = [factors]
    return {f: np.sort(y[f].astype(str).dropna().unique()) for f in factors}

def _combo_mask(y, factors, combo):
    m = np.ones(len(y), dtype=bool)
    for f, v in zip(factors, combo):
        s = y[f]
        if np.issubdtype(s.dtype, np.number):
            m &= (s.values == v)
        else:
            m &= (s.astype(str).values == str(v))
    return m

def cv_avg_cond(X, y, factors, levels, drop_missing=False):
    if isinstance(factors, str):
        factors = [factors]
    combos = list(itertools.product(*[levels[f] for f in factors]))
    X_avg = []
    for combo in combos:
        idx = _combo_mask(y, factors, combo)
        if not idx.any():
            if drop_missing:
                continue
            raise ValueError(f"Missing combo: {dict(zip(factors, combo))}")
        X_avg.append(np.nanmean(X[idx], axis=0))
    return np.stack(X_avg, axis=0), combos

def anchors_from_Z(Z, y, factors, levels, drop_missing=False):
    if isinstance(factors, str):
        factors = [factors]
    combos = list(itertools.product(*[levels[f] for f in factors]))
    A = []
    for combo in combos:
        idx = _combo_mask(y, factors, combo)
        if not idx.any():
            if drop_missing:
                continue
            raise ValueError(f"Missing combo: {dict(zip(factors, combo))}")
        A.append(np.nanmean(Z[idx], axis=0))
    return np.stack(A, axis=0).reshape(-1, Z.shape[-1])

def cv_pca(X, y, pca, folds, factors, X_raw=None,
           epoch=None, scale=None, stage='Expert',
           correct=True, context=None, show_pbar=True,
           center_on="trialtime", align_folds=False):
    """
    Cross-validated PCA with optional Procrustes alignment across folds.

    Parameters
    ----------
    X : (n_trials, n_neurons, n_time) — data for PCA fitting
    y : DataFrame
    X_raw : (n_trials, n_neurons, n_time) — raw data for test-set projection
    epoch : time indices for fitting; None = full trial
    scale : if not None, apply CenterScaler with scaling
    stage : learning stage to include in clean mask
    correct : if True, only use correct trials for fitting
    """
    if isinstance(factors, str):
        factors = [factors]

    def flat_trial_time(A):
        # (trials, neurons, time) -> (trials*time, neurons)
        return A.transpose(0, 2, 1).reshape(-1, A.shape[1])

    # Build clean mask
    m_clean = (y.laser == 0)
    if stage is not None:
        m_clean &= (y.learning == stage)
    if correct:
        m_clean &= (y.performance == 1)
    if context is not None:
        m_clean &= (y.tasks == context)

    if X_raw is None:
        X_raw = X.copy()

    Xc = X[m_clean]
    Xt = X_raw[m_clean]
    yc = y.loc[m_clean].reset_index(drop=True)

    # Optional scaling (fit on all clean data for perturbed projection)
    sc_all = None
    if scale is not None:
        sc_all = CenterScaler(scale=True)
        Xc = sc_all.fit_transform(Xc)

    levels = get_levels(yc, factors)

    # Fit canonical PCA on all clean data (for reference + perturbed projection)
    Xc_ep = Xc if epoch is None else Xc[..., epoch]
    X_avg_all, _ = cv_avg_cond(Xc_ep, yc, factors, levels, drop_missing=True)
    X_avg_flat = flat_trial_time(X_avg_all)

    if center_on == "trialtime":
        X_mean_all = np.nanmean(flat_trial_time(Xc_ep), axis=0, keepdims=True)
    else:
        X_mean_all = np.nanmean(X_avg_flat, axis=0, keepdims=True)

    pca_all = clone(pca).fit(X_avg_flat - X_mean_all)
    W_ref = pca_all.components_
    evr_all = pca_all.explained_variance_ratio_

    # Reference anchors for Procrustes alignment
    Zc_ep = pca_all.transform(flat_trial_time(Xc_ep) - X_mean_all)
    Zc_ep = Zc_ep.reshape(Xc_ep.shape[0], Xc_ep.shape[-1], -1)
    A_ref = anchors_from_Z(Zc_ep, yc, factors, levels, drop_missing=True)

    # Build stratification for CV splits
    strata = (yc["odor_pair"].astype(str) + "_" + yc["day"].astype(str)
              + "_" + yc["tasks"].astype(str)).values

    # Get iterator
    n_total = folds.get_n_splits(Xc, strata)
    it = folds.split(Xc, strata)
    if show_pbar:
        name = y.mouse.unique()[0] if "mouse" in y.columns else ""
        it = tqdm(it, total=n_total, desc=name)

    Z_folds, y_folds, evr_folds = [], [], []

    for train, test in it:
        X_train, y_train = Xc[train], yc.iloc[train].reset_index(drop=True)
        X_test = Xt[test]
        y_test = yc.iloc[test].reset_index(drop=True)

        if scale is not None:
            sc = CenterScaler(scale=True)
            X_train = sc.fit_transform(X_train)
            X_test = sc.transform(X_test)

        X_train_ep = X_train if epoch is None else X_train[..., epoch]
        try:
            X_avg_tr, _ = cv_avg_cond(X_train_ep, y_train, factors, levels, drop_missing=True)
        except ValueError:
            # Missing condition combo in this fold — skip
            continue

        X_avg_fl = flat_trial_time(X_avg_tr)

        if center_on == "trialtime":
            X_mean = np.nanmean(flat_trial_time(X_train_ep), axis=0, keepdims=True)
        else:
            X_mean = np.nanmean(X_avg_fl, axis=0, keepdims=True)

        pca_fold = clone(pca).fit(X_avg_fl - X_mean)
        evr_folds.append(pca_fold.explained_variance_ratio_)

        W = pca_fold.components_
        Z_test = flat_trial_time(X_test) @ W.T
        Z_test = Z_test.reshape(X_test.shape[0], X_test.shape[-1], -1)

        if align_folds:
            Z_tr_ep = pca_fold.transform(flat_trial_time(X_train_ep) - X_mean)
            Z_tr_ep = Z_tr_ep.reshape(X_train_ep.shape[0], X_train_ep.shape[-1], -1)
            A_train = anchors_from_Z(Z_tr_ep, y_train, factors, levels, drop_missing=True)
            R, _ = orthogonal_procrustes(A_train, A_ref)
            Z_test = Z_test @ R

        Z_folds.append(Z_test)
        y_folds.append(y_test)

    # Project perturbed (laser-on) trials using full-data PCA
    m_pert = ~m_clean
    if np.any(m_pert):
        Xp = X_raw[m_pert]
        yp = y.loc[m_pert].reset_index(drop=True)
        if sc_all is not None:
            Xp = sc_all.transform(Xp)
        Zp = flat_trial_time(Xp) @ W_ref.T
        Zp = Zp.reshape(Xp.shape[0], Xp.shape[-1], -1)
        Z_all = np.concatenate([np.concatenate(Z_folds, 0), Zp], 0)
        y_all_out = pd.concat([pd.concat(y_folds, ignore_index=True), yp], ignore_index=True)
    else:
        Z_all = np.concatenate(Z_folds, 0)
        y_all_out = pd.concat(y_folds, ignore_index=True)

    return Z_all, y_all_out, W_ref, np.array(evr_folds, dtype=object)

def participation_ratio(eigenvalues):
    """PR = (sum lambda_i)^2 / sum(lambda_i^2)."""
    ev = np.asarray(eigenvalues, dtype=float)
    ev = ev[ev > 0]
    return (ev.sum()) ** 2 / (ev ** 2).sum()

def epoch_xy(X, bins, x_idx=0, y_idx=1):
    """Mean position in a 2D plane over a time epoch."""
    x = np.nanmean(X[:, x_idx, bins], axis=-1)
    y = np.nanmean(X[:, y_idx, bins], axis=-1)
    return x, y

def traj_xy(X, bins=None, x_idx=0, y_idx=1):
    if bins is None:
        return X[:, x_idx, :], X[:, y_idx, :]
    return X[:, x_idx, bins], X[:, y_idx, bins]

def subspace_angle_vectors(w1, w2):
    """Angle in degrees between two vectors."""
    w1n = w1 / (np.linalg.norm(w1) + 1e-12)
    w2n = w2 / (np.linalg.norm(w2) + 1e-12)
    return np.rad2deg(np.arccos(np.clip(np.abs(w1n @ w2n), 0, 1)))

def distance_to_boundary(X_latent, y, bins, x_idx=0, y_idx=1,
                         label_col='choice', stage='Expert'):
    """
    Fit logistic boundary on expert laser-off correct trials,
    return signed distance per trial (positive = correct side).
    """
    mask_fit = ((y.laser == 0) & (y.learning == stage) &
                (y.performance == 1)).values
    x_fit, y_fit_coord = epoch_xy(X_latent[mask_fit], bins, x_idx, y_idx)
    features = np.column_stack([x_fit, y_fit_coord])
    labels = y.loc[mask_fit, label_col].values.astype(int)

    clf = LogisticRegression(solver='lbfgs', max_iter=1000).fit(features, labels)

    # Signed distance for ALL trials
    x_all, y_all = epoch_xy(X_latent, bins, x_idx, y_idx)
    feat_all = np.column_stack([x_all, y_all])
    w = clf.coef_[0]
    b = clf.intercept_[0]
    raw_dist = (feat_all @ w + b) / np.linalg.norm(w)

    # Sign: positive if prediction matches true label
    pred_label = (raw_dist > 0).astype(int)
    true_label = y[label_col].values.astype(int)
    signed = np.where(pred_label == true_label, np.abs(raw_dist), -np.abs(raw_dist))
    return signed, clf

def flow_field_midpoint(x, y, dt=1.0, bins=25, xrange=None, yrange=None, min_count=1):
    """Bin-averaged velocity field using segment midpoints."""
    x = np.asarray(x, dtype=float)
    y = np.asarray(y, dtype=float)
    n_trials, n_time = x.shape

    u = (x[:, 1:] - x[:, :-1]) / dt
    v = (y[:, 1:] - y[:, :-1]) / dt
    xm = 0.5 * (x[:, 1:] + x[:, :-1])
    ym = 0.5 * (y[:, 1:] + y[:, :-1])

    if xrange is None: xrange = (np.nanmin(xm), np.nanmax(xm))
    if yrange is None: yrange = (np.nanmin(ym), np.nanmax(ym))

    if isinstance(bins, int):
        nx = ny = bins
    else:
        nx, ny = bins

    xedges = np.linspace(xrange[0], xrange[1], nx + 1)
    yedges = np.linspace(yrange[0], yrange[1], ny + 1)

    xf, yf = xm.ravel(), ym.ravel()
    uf, vf = u.ravel(), v.ravel()

    # Remove NaNs
    finite = np.isfinite(xf) & np.isfinite(yf) & np.isfinite(uf) & np.isfinite(vf)
    xf, yf, uf, vf = xf[finite], yf[finite], uf[finite], vf[finite]

    ix = np.searchsorted(xedges, xf, side="right") - 1
    iy = np.searchsorted(yedges, yf, side="right") - 1
    valid = (ix >= 0) & (ix < nx) & (iy >= 0) & (iy < ny)
    ix, iy, uf, vf = ix[valid], iy[valid], uf[valid], vf[valid]

    count = np.zeros((nx, ny), int)
    usum = np.zeros((nx, ny))
    vsum = np.zeros((nx, ny))
    np.add.at(count, (ix, iy), 1)
    np.add.at(usum, (ix, iy), uf)
    np.add.at(vsum, (ix, iy), vf)

    U = np.where(count >= min_count, usum / np.maximum(count, 1), np.nan)
    V = np.where(count >= min_count, vsum / np.maximum(count, 1), np.nan)
    return xedges, yedges, U, V, count

def compute_speed_field(x, y, bins=25, dt=1.0, smooth=1.0, min_count=5):
    xedges, yedges, U, V, C = flow_field_midpoint(x, y, dt=dt, bins=bins)
    speed = np.sqrt(U**2 + V**2)
    speed = np.where(C >= min_count, speed, np.nan)
    if smooth and smooth > 0:
        U = gaussian_filter(np.nan_to_num(U), smooth)
        V = gaussian_filter(np.nan_to_num(V), smooth)
        s0 = gaussian_filter(np.nan_to_num(speed), smooth)
        n0 = gaussian_filter(np.isfinite(speed).astype(float), smooth)
        speed = np.where(n0 > 0, s0 / n0, np.nan)
    return xedges, yedges, speed, U, V, C

def occupancy_map(x, y, bins=25, zlim=None, smooth=1.0):
    xf, yf = x.flatten(), y.flatten()
    finite = np.isfinite(xf) & np.isfinite(yf)
    xf, yf = xf[finite], yf[finite]
    if zlim is None:
        zlim = np.nanmax(np.abs(np.r_[xf, yf]))
    xedges = np.linspace(-zlim, zlim, bins + 1)
    yedges = np.linspace(-zlim, zlim, bins + 1)
    H, _, _ = np.histogram2d(xf, yf, bins=[xedges, yedges])
    H = H.T
    if smooth and smooth > 0:
        H = gaussian_filter(H, smooth)
    mx = np.nanmax(H)
    H = H / mx if mx > 0 else H
    return xedges, yedges, H

def ccgd_sample_across_context(X, y, bins, decode_col='sample_odor',
                                context_col='odor_pair', n_splits=5, n_repeats=5):
    """
    Cross-condition generalization: train sample decoder in one context,
    test in another. Uses odor_pair to define contexts since all trials
    are 'Dual' task.

    Contexts: pairs {0,1} (sample A: AC, AD) vs {2,3} (sample B: BD, BC)
    won't work — instead split by test odor (C vs D context).
    """
    mask = ((y.laser == 0) & (y.learning == 'Expert') & (y.performance == 1)).values
    X_m = X[mask]
    y_m = y.loc[mask].reset_index(drop=True)

    # Features: mean over epoch bins
    feat = np.nanmean(X_m[:, :3, bins], axis=-1)
    labels = y_m[decode_col].values.astype(int)

    # Context: test odor (0=C, 1=D) — each test context contains both samples
    contexts = y_m['test_odor'].values.astype(int)
    ctx_vals = np.unique(contexts)

    if len(ctx_vals) < 2:
        return np.nan, np.nan

    within_scores, cross_scores = [], []

    for rep in range(n_repeats):
        for ctx_tr, ctx_te in [(ctx_vals[0], ctx_vals[1]),
                                (ctx_vals[1], ctx_vals[0])]:
            m_tr = (contexts == ctx_tr)
            m_te = (contexts == ctx_te)

            if m_tr.sum() < 10 or m_te.sum() < 10:
                continue

            # Check both classes present
            if len(np.unique(labels[m_tr])) < 2 or len(np.unique(labels[m_te])) < 2:
                continue

            clf = LinearSVC(max_iter=5000, dual='auto')
            clf.fit(feat[m_tr], labels[m_tr])
            cross_scores.append(clf.score(feat[m_te], labels[m_te]))

            # Within: stratified CV on train context
            n_min = min(np.bincount(labels[m_tr]))
            if n_min >= n_splits:
                cv = StratifiedKFold(n_splits=n_splits, shuffle=True,
                                     random_state=rep)
                scores = cross_val_score(
                    LinearSVC(max_iter=5000, dual='auto'),
                    feat[m_tr], labels[m_tr], cv=cv
                )
                within_scores.extend(scores.tolist())

    within = np.mean(within_scores) if within_scores else np.nan
    cross = np.mean(cross_scores) if cross_scores else np.nan
    return within, cross
