                  means = model.named_steps["scaler"].mean_
                  scales = model.named_steps["scaler"].scale_

def _get_base_estimator(estimator):
    est = getattr(estimator, "best_estimator_", estimator)
    if hasattr(est, "named_steps"):  # Pipeline
        return list(est.named_steps.values())[-1]
    return est


def overlaps_scorer(estimator, X_test, y_test, IF_SIGN=0, eps=1e-6):
    clf = getattr(estimator, "best_estimator_", estimator)

    if not hasattr(clf, "decision_function"):
        raise AttributeError("Resolved estimator has no decision_function method.")

    model = _get_base_estimator(estimator)

    if not hasattr(model, "coef_"):
        raise AttributeError("Underlying model has no coef_ attribute.")

    norm_w = np.linalg.norm(np.ravel(model.coef_)) + eps
    scores = clf.decision_function(X_test) / norm_w

    if IF_SIGN:
        scores = (2 * y_test - 1) * scores

    return np.nanmean(scores)

def bootstrap_ci(data, n_bootstrap=1000, ci=95):
    bootstrapped_means = np.array([np.mean(np.random.choice(data, size=len(data))) for _ in range(n_bootstrap)])
    lower_bound = np.percentile(bootstrapped_means, (100-ci)/2)
    upper_bound = np.percentile(bootstrapped_means, 100 - (100-ci)/2)
    return lower_bound, upper_bound
#+end_src
