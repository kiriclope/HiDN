def pad_list(arrays, axis=0, max_len=None):
    """
    Pads a list of arrays along the specified axis with NaNs so all have the same size along that axis.
    Returns a list of padded arrays.
    """
    # Find maximum size along specified axis
    if max_len is None:
        max_len = max(arr.shape[axis] for arr in arrays)

    padded = []
    for arr in arrays:
        pad_width = [(0, 0)] * arr.ndim


        n_pad = max_len - arr.shape[axis]

        if n_pad > 0:
            pad_width[axis] = (0, n_pad)
            arr_padded = np.pad(arr, pad_width, mode='constant', constant_values=np.nan)
        else:
            arr_padded = arr
        padded.append(arr_padded)

    return padded

def overlaps_scorer(estimator, X, y, IF_SIGN=0, eps=1e-6):
    """
    Custom scorer called by MNE on a single fitted estimator, typically a Pipeline.
    """
    if not hasattr(estimator, "decision_function"):
        raise AttributeError("Estimator has no decision_function method.")

    dec = np.asarray(estimator.decision_function(X))

    coef = _get_coef_single(estimator)
    norm = np.linalg.norm(np.ravel(coef)) + eps
    dec_norm = dec / norm

    if IF_SIGN:
        sgn = 2 * np.asarray(y) - 1
        dec_norm = sgn * dec_norm

    return np.nanmean(dec_norm)

    """
    For SlidingEstimator or GeneralizingEstimator fitted on time-resolved data.

    norms = []
    for est_t in time_estimator.estimators_:
        final_est = get_final_estimator(est_t)
        coef = np.asarray(final_est.coef_)
        norms.append(np.linalg.norm(coef.ravel()) + eps)

import numpy as np

def get_rawspace_params_timewise(est):
    """
    Extract raw-space linear weights/intercepts from a fitted
    SlidingEstimator or GeneralizingEstimator.

    Assumes X has shape (n_trials, n_features, n_times).
    Returns
    -------
    ws : (n_train_times, n_features)
    bs : (n_train_times,)
    """
    ws, bs = [], []

    for est_t in est.estimators_:
        if isinstance(est_t, Pipeline):
            clf = est_t.named_steps["model"]
            scaler = est_t.named_steps.get("scaler", None)
        else:
            clf = est_t
            scaler = None

        w = np.asarray(clf.coef_).squeeze()
        b = np.asarray(clf.intercept_).squeeze()

        if scaler is not None:
            mean = getattr(scaler, "mean_", None)
            scale = getattr(scaler, "scale_", None)

            if mean is None:
                mean = 0.0
            if scale is None:
                scale = 1.0

            w_orig = w.copy()
            w = w / scale
            b = b - np.sum(w_orig * mean / scale)

        ws.append(w)
        bs.append(b)

    return np.asarray(ws), np.asarray(bs)


def get_coef_norms_timewise(est, eps=1e-6):
    """Norm of transformed-space coefficients for each train-time estimator."""
    norms = []
    for est_t in est.estimators_:
        clf = est_t.named_steps["model"] if isinstance(est_t, Pipeline) else est_t
        coef = np.asarray(clf.coef_).squeeze()
        norms.append(np.linalg.norm(coef.ravel()) + eps)
    return np.asarray(norms)



def get_norms(est, raw=False, eps=1e-6):
    """Norms matching the chosen decision space."""
    if raw:
        ws, _ = get_rawspace_params_timewise(est)
        return np.linalg.norm(ws, axis=1) + eps
    return get_coef_norms_timewise(est, eps=eps)

def get_decision_values(est, X, raw=False, remove_intercept=True):
    ws, bs = get_rawspace_params_timewise(est)

    if remove_intercept:
        bs = np.zeros_like(bs)

    if est.__class__.__name__ == "SlidingEstimator":
        return np.einsum("nft,tf->nt", X, ws) + bs[None, :]
    return np.einsum("nfu,tf->ntu", X, ws) + bs[None, :, None]

def postprocess_decision(dec, norms, y=None, signed=False):
    """
    Normalize by weight norm and optionally sign by class label.
    Assumes binary y coded as 0/1.
    """
    dec = np.asarray(dec)

    if dec.ndim == 3:
        dec = dec / norms[None, :, None]
        if signed:
            dec = (2 * np.asarray(y) - 1)[:, None, None] * dec
    elif dec.ndim == 2:
        dec = dec / norms[None, :]
        if signed:
            dec = (2 * np.asarray(y) - 1)[:, None] * dec
    else:
        raise ValueError(f"Unexpected decision shape: {dec.shape}")

    return dec

def cross_val_score(
    X,
    y_df,
    estimator,
    target="sample",
    cv=None,
    stage="Expert",
    signed=False,
    raw=False,
    correct=True,
):
    """
    Cross-validated single-trial decision values.

    Parameters
    ----------
    X : array, shape (n_trials, n_features, n_times)
    y_df : DataFrame
    estimator : unfitted SlidingEstimator or GeneralizingEstimator
    target : str
        Column in y_df containing binary labels.
    cv : splitter or None
    stage : str
        Used to define the within set:
        laser == 0, learning == stage, performance == 1
    signed : bool
        If True, multiply decisions by +/-1 according to labels.
    raw : bool
        If True, compute projections in raw feature space.
        If False, use estimator.decision_function in transformed space.

    Returns
    -------
    scores : array
        within scores stacked with mean cross scores
    y_cv : DataFrame
        metadata aligned with scores
    """
    start = perf_counter()

    if cv is None:
        cv = StratifiedKFold(n_splits=5, shuffle=True)

    idx_correct = True
    if correct:
        idx_correct = (y_df.performance == 1)

    m_within = (
        (y_df.laser == 0)
        & (y_df.learning == stage)
        & idx_correct
    )

    m_cross = ~m_within

    X_within = X[m_within]
    y_within = y_df.loc[m_within].reset_index(drop=True).copy()
    y_within["labels"] = y_within[target].to_numpy()
    y_within_labels = y_within["labels"].to_numpy()

    X_cross = X[m_cross]
    y_cross = y_df.loc[m_cross].reset_index(drop=True).copy()
    y_cross["labels"] = y_cross[target].to_numpy()
    y_cross_labels = y_cross["labels"].to_numpy()

    print("X_within", X_within.shape, "y_within", y_within.shape)
    print("X_cross", X_cross.shape, "y_cross", y_cross.shape)

    strata = (
        y_within["odor_pair"].astype(str) + "_"
        + y_within["day"].astype(str) + "_"
        + y_within["tasks"].astype(str)
    ).to_numpy()

    within_scores = []
    y_cv_within = []
    cross_scores = []

    for train_idx, test_idx in cv.split(X_within, strata):
        X_train, X_test = X_within[train_idx], X_within[test_idx]
        y_train, y_test = y_within_labels[train_idx], y_within_labels[test_idx]

        est = clone(estimator)
        est.fit(X_train, y_train)

        norms = get_norms(est, raw=raw)

        within_dec = get_decision_values(est, X_test, raw=raw)
        cross_dec = get_decision_values(est, X_cross, raw=raw)

        within_scores.append(
            postprocess_decision(within_dec, norms, y=y_test, signed=signed)
        )
        cross_scores.append(
            postprocess_decision(cross_dec, norms, y=y_cross_labels, signed=signed)
        )

        y_cv_within.append(y_within.iloc[test_idx].reset_index(drop=True))

    within_scores = np.vstack(within_scores)
    y_cv_within = pd.concat(y_cv_within, axis=0, ignore_index=True)

    cross_scores = np.mean(np.stack(cross_scores, axis=0), axis=0)
    y_cv_cross = y_cross

    print("within_score", within_scores.shape, "y_within", y_cv_within.shape)
    print("cross_score", cross_scores.shape, "y_cross", y_cv_cross.shape)

    scores = np.vstack([within_scores, cross_scores])
    y_cv = pd.concat([y_cv_within, y_cv_cross], axis=0, ignore_index=True)

    end = perf_counter()
    print("Elapsed (with compilation) = %dh %dm %ds" % convert_seconds(end - start))

    return scores, y_cv

def get_rawspace_params_timewise(est):
    """
    Back-project weights through the scaler to raw feature space.

    Returns
    -------
    ws : (n_train_times, n_features)
    bs : (n_train_times,)
    """
    ws, bs = [], []

    for est_t in est.estimators_:
        if isinstance(est_t, Pipeline):
            clf = est_t.named_steps["model"]
            scaler = est_t.named_steps.get("scaler", None)
        else:
            clf = est_t
            scaler = None

        w = np.asarray(clf.coef_).ravel()
        b = np.asarray(clf.intercept_).item()

        if scaler is not None:
            mean = getattr(scaler, "mean_", None)
            scale = getattr(scaler, "scale_", None)

            if mean is None:
                mean = 0.0
            if scale is None:
                scale = 1.0

            w = w / scale
            b = b - np.dot(w, mean)  # note: w is already rescaled here

        ws.append(w)
        bs.append(b)

    return np.asarray(ws), np.asarray(bs)


def get_decision_values(est, X, remove_intercept=True):
    """
    Project trials onto the raw-space discriminant axis.

    Parameters
    ----------
    est : fitted SlidingEstimator or GeneralizingEstimator
    X : (n_trials, n_features, n_times)
    remove_intercept : bool
        If True, pure projection onto w (no intercept).

    Returns
    -------
    dec : (n_trials, n_times) or (n_trials, n_train_times, n_test_times)
    """
    ws, bs = get_rawspace_params_timewise(est)

    if remove_intercept:
        bs = np.zeros_like(bs)

    if est.__class__.__name__ == "SlidingEstimator":
        return np.einsum("nft,tf->nt", X, ws) + bs[None, :]

    return np.einsum("nfu,tf->ntu", X, ws) + bs[None, :, None]


def get_norms_timewise(est):
    """Raw-space weight norms per train-time."""
    ws, _ = get_rawspace_params_timewise(est)
    return np.linalg.norm(ws, axis=1)


def postprocess_decision(dec, norms, y=None, signed=False, eps=1e-6):
    """
    Normalize by weight norm, optionally sign by label (0/1 coding).
    """
    norms = norms + eps

    if dec.ndim == 2:  # (n_trials, n_times)
        dec = dec / norms[None, :]
        if signed:
            dec = (2 * np.asarray(y) - 1)[:, None] * dec
    elif dec.ndim == 3:  # (n_trials, n_train_times, n_test_times)
        dec = dec / norms[None, :, None]
        if signed:
            dec = (2 * np.asarray(y) - 1)[:, None, None] * dec
    else:
        raise ValueError(f"Unexpected shape: {dec.shape}")

    return dec

def ccgd_val_score(
    X,
    y_df,
    estimator,
    target="sample",
    cv=None,
    stage="Expert",
    context="DPA",
    signed=False,
    correct=False,
    remove_intercept=False,
):
    """
    Cross-validated single-trial decision values with cross-condition
    generalization.
    """
    start = perf_counter()

    if cv is None:
        cv = StratifiedKFold(n_splits=5, shuffle=True)

    idx_correct = True
    if correct:
        idx_correct = (y_df.performance == 1) & (
            (y_df.tasks == "DPA") | (y_df.odr_perf == 1)
        )

    m_within = (
        (y_df.laser == 0)
        & (y_df.learning == stage)
        & (y_df.tasks == context)
        & idx_correct
    )
    m_cross = ~m_within

    X_within = X[m_within]
    y_within = y_df.loc[m_within].reset_index(drop=True).copy()
    y_within["labels"] = y_within[target].to_numpy()
    y_within_labels = y_within["labels"].to_numpy()

    X_cross = X[m_cross]
    y_cross = y_df.loc[m_cross].reset_index(drop=True).copy()
    y_cross["labels"] = y_cross[target].to_numpy()
    y_cross_labels = y_cross["labels"].to_numpy()

    print(f"X_within {X_within.shape}  y_within {y_within.shape}")
    print(f"X_cross  {X_cross.shape}  y_cross  {y_cross.shape}")

    strata = (
        y_within["odor_pair"].astype(str)
        + "_"
        + y_within["day"].astype(str)
    ).to_numpy()

    within_scores = []
    y_cv_within = []
    cross_scores = []

    for train_idx, test_idx in cv.split(X_within, strata):
        X_train, X_test = X_within[train_idx], X_within[test_idx]
        y_train = y_within_labels[train_idx]
        y_test = y_within_labels[test_idx]

        est = clone(estimator)
        est.fit(X_train, y_train)

        norms = get_norms_timewise(est)

        within_dec = get_decision_values(est, X_test, remove_intercept=remove_intercept)
        cross_dec = get_decision_values(est, X_cross, remove_intercept=remove_intercept)

        within_scores.append(
            postprocess_decision(within_dec, norms, y=y_test, signed=signed)
        )
        cross_scores.append(
            postprocess_decision(cross_dec, norms, y=y_cross_labels, signed=signed)
        )

        y_cv_within.append(y_within.iloc[test_idx].reset_index(drop=True))

    within_scores = np.vstack(within_scores)
    y_cv_within = pd.concat(y_cv_within, axis=0, ignore_index=True)

    cross_scores = np.mean(np.stack(cross_scores, axis=0), axis=0)
    y_cv_cross = y_cross

    print(f"within_scores {within_scores.shape}  y_cv_within {y_cv_within.shape}")
    print(f"cross_scores  {cross_scores.shape}  y_cv_cross  {y_cv_cross.shape}")

    scores = np.vstack([within_scores, cross_scores])
    y_cv = pd.concat([y_cv_within, y_cv_cross], axis=0, ignore_index=True)

    end = perf_counter()
    print("Elapsed = %dh %dm %ds" % convert_seconds(end - start))

    return scores, y_cv

# delay bins after binning-by-2
# if options["bins_DELAY"] exists in original 84-bin space, divide by 2
delay_bins = options["bins_LD"]

rows = []
for (mouse, context, task, stage), g in df.groupby(
    ["mouse", "context", "tasks", "learning"], sort=False
):
    T = np.stack(g["traj"].to_numpy(), axis=0)   # (n_trials, n_time)

    # mean decoding trajectory across trials
    mean_traj = np.nanmean(T, axis=0)            # (n_time,)

    # average strength in delay period
    val = np.nanmean(mean_traj[delay_bins])

    rows.append({
        "mouse": mouse,
        "context": context,
        "task": task,
        "stage": stage,
        "value": val,
        "n_trials": len(g),
    })

plot_df = pd.DataFrame(rows)
