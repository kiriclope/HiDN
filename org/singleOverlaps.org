def pad_list(arrays, axis=0, max_len=None):
    """
    Pads a list of arrays along the specified axis with NaNs so all have the same size along that axis.
    Returns a list of padded arrays.
    """
    # Find maximum size along specified axis
    if max_len is None:
        max_len = max(arr.shape[axis] for arr in arrays)

    padded = []
    for arr in arrays:
        pad_width = [(0, 0)] * arr.ndim


        n_pad = max_len - arr.shape[axis]

        if n_pad > 0:
            pad_width[axis] = (0, n_pad)
            arr_padded = np.pad(arr, pad_width, mode='constant', constant_values=np.nan)
        else:
            arr_padded = arr
        padded.append(arr_padded)

    return padded

import numpy as np
import pandas as pd
from time import perf_counter
from sklearn.base import clone
from sklearn.pipeline import Pipeline
from sklearn.preprocessing import StandardScaler

from sklearn.linear_model import LogisticRegressionCV, LogisticRegression
from mne.decoding import SlidingEstimator, GeneralizingEstimator

def get_estimator(
        clf='logcv',
        scoring="accuracy",
        mode="generalizing",
        scaler=None,
        l1_ratio=0,
        C=0.1,
        Cs = np.logspace(-4, 4, 10),
        n_jobs=-1,
):

    if l1_ratio in [0, 1]:
        solver = 'liblinear'
    else:
        solver = 'saga'

    if clf=='logcv':
        clf = LogisticRegressionCV(cv=5, solver=solver, class_weight="balanced", n_jobs=None, l1_ratios=(l1_ratio,), Cs=Cs)
    else:
        clf = LogisticRegression(solver=solver, class_weight="balanced", n_jobs=None, l1_ratio=l1_ratio)

    pipeline = []
    if scaler is not None:
        if scaler == "standard":
            pipeline.append(("scaler", StandardScaler()))
        elif scaler == "center":
            pipeline.append(("scaler", StandardScaler(with_std=False)))

    pipeline.append(("model", clf))
    pipe = Pipeline(pipeline)

    if mode == "sliding":
        estimator = SlidingEstimator(pipe, scoring=scoring, n_jobs=n_jobs, verbose=False)
    elif mode=="generalizing":
        estimator = GeneralizingEstimator(pipe, scoring=scoring, n_jobs=n_jobs, verbose=False)
    else:
        estimator = pipe

    return estimator

def get_space_params_timewise(est, raw=False):
    """
    Back-project weights through the scaler to raw feature space.

    Returns
    -------
    ws : (n_train_times, n_features)
    bs : (n_train_times,)
    """
    ws, bs = [], []

    for est_t in est.estimators_:
        if isinstance(est_t, Pipeline):
            clf = est_t.named_steps["model"]
            scaler = est_t.named_steps.get("scaler", None)
        else:
            clf = est_t
            scaler = None

        w = np.asarray(clf.coef_).ravel()
        b = np.asarray(clf.intercept_).item()

        if scaler is not None:
            mean = getattr(scaler, "mean_", None)
            scale = getattr(scaler, "scale_", None)

            if mean is None:
                mean = 0.0
            if scale is None:
                scale = 1.0

            if raw:
                w = w / scale
                b = b - np.dot(w, mean)  # note: w is already rescaled here

        ws.append(w)
        bs.append(b)

    return np.asarray(ws), np.asarray(bs)


def get_decision_values(est, X, raw=False, remove_intercept=True):
    """
    Project trials onto the raw-space discriminant axis.

    Parameters
    ----------
    est : fitted SlidingEstimator or GeneralizingEstimator
    X : (n_trials, n_features, n_times)
    remove_intercept : bool
        If True, pure projection onto w (no intercept).

    Returns
    -------
    dec : (n_trials, n_times) or (n_trials, n_train_times, n_test_times)
    """

    ws, bs = get_space_params_timewise(est, raw)

    if remove_intercept:
        bs = np.zeros_like(bs)

    if est.__class__.__name__ == "SlidingEstimator":
        return np.einsum("nft,tf->nt", X, ws) + bs[None, :]

    return np.einsum("nfu,tf->ntu", X, ws) + bs[None, :, None]


def get_norms_timewise(est, raw):
    """Raw-space weight norms per train-time."""
    ws, _ = get_space_params_timewise(est, raw)
    return np.linalg.norm(ws, axis=1)


def postprocess_decision(dec, norms, y=None, signed=False, eps=1e-6):
    """
    Normalize by weight norm, optionally sign by label (0/1 coding).
    """
    norms = norms + eps

    if dec.ndim == 2:  # (n_trials, n_times)
        dec = dec / norms[None, :]
        if signed:
            dec = (2 * np.asarray(y) - 1)[:, None] * dec
    elif dec.ndim == 3:  # (n_trials, n_train_times, n_test_times)
        dec = dec / norms[None, :, None]
        if signed:
            dec = (2 * np.asarray(y) - 1)[:, None, None] * dec
    else:
        raise ValueError(f"Unexpected shape: {dec.shape}")

    return dec

    X,
    y_df,
    estimator,
    selector=None,

    if strata:
        strata = (
            y_within["odor_pair"].astype(str)
            + "_"
            + y_within["odr_choice"].astype(str)
            + "_"
            + y_within["choice"].astype(str)
            + "_"
            + y_within["day"].astype(str)
        ).to_numpy()
    else:
        strata = y_within[target].astype(str).to_numpy()

    within_probas, within_dfs = [], []
    y_cv_within = []
    cross_probas, cross_dfs = [], []

    for train_idx, test_idx in cv.split(X_within, strata):
        X_train, X_test = X_within[train_idx], X_within[test_idx]
        y_train = y_within_labels[train_idx]
        y_test = y_within_labels[test_idx]
        y_cv_within.append(y_within.iloc[test_idx].reset_index(drop=True))

        est = clone(estimator)

        if fit_param_epoch:
            sel = clone(selector)
            sel.fit(X_train[..., bins_epoch].mean(-1), y_train)

            best_C = sel.named_steps["model"].C_[0]
            est.base_estimator.named_steps["model"].C = best_C

        est.fit(X_train, y_train)

        # decision function
        norms = get_norms_timewise(est, raw)

        within_dec = get_decision_values(est, X_test, raw=raw, remove_intercept=remove_intercept)
        within_dec = postprocess_decision(within_dec, norms, y=y_test, signed=signed)
        within_dfs.append(within_dec)

        cross_dec = get_decision_values(est, X_cross, raw=raw, remove_intercept=remove_intercept)
        cross_dec = postprocess_decision(cross_dec, norms, y=y_cross_labels, signed=signed)
        cross_dfs.append(cross_dec)

        # probas
        within_proba = est.predict_proba(X_test)[..., 1]
        within_proba = y_test[:, None, None] * within_proba + (1 - y_test[:, None, None]) * (1 - within_proba)
        within_probas.append(within_proba)

        cross_proba = est.predict_proba(X_cross)[..., 1]
        cross_proba = y_cross_labels[:, None, None] * cross_proba + (1 - y_cross_labels[:, None, None]) * (1 - cross_proba)
        cross_probas.append(cross_proba)

    within_probas = np.vstack(within_probas)
    within_dfs = np.vstack(within_dfs)

    y_cv_within = pd.concat(y_cv_within, axis=0, ignore_index=True)

    cross_probas = np.mean(np.stack(cross_probas, axis=0), axis=0)
    cross_dfs = np.mean(np.stack(cross_dfs, axis=0), axis=0)

    y_cv_cross = y_cross

    print(f"within_probas {within_probas.shape}  y_cv_within {y_cv_within.shape}")
    print(f"cross_probas  {cross_probas.shape}  y_cv_cross  {y_cv_cross.shape}")

    probas = np.vstack([within_probas, cross_probas])
    dfs = np.vstack([within_dfs, cross_dfs])

    y_cv = pd.concat([y_cv_within, y_cv_cross], axis=0, ignore_index=True)

    end = perf_counter()
    print("Elapsed = %dh %dm %ds" % convert_seconds(end - start))

    return probas, dfs, y_cv
